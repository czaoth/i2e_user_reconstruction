% Custom command for including the source of a media file.
% The first parameter is the source of the media file.
\newcommand{\customSource}[1]
{
	\hspace*{15pt}\hbox{\tiny Quelle:\thinspace{\tiny\itshape #1}}
}

% Custom command for including page filling images.
% First parameter is the image files path (including its extension).
% The second parameter is the description of the image.
% The third parameter is the source of the image.
\newcommand{\customImage}[3] 
{
	\begin{figure}[h!]
		\begin{center}		
			\includegraphics[width=\defaultImageWidth\textwidth, keepaspectratio]{#1}
			\caption{#2}
			\customSource{#3}
		\end{center}
	\end{figure}
}
